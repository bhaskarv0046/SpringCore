create or replace PROCEDURE SPRING_AUTHENTICATION 
(
  UNAME IN VARCHAR2 
, PARAM1 IN VARCHAR2 
, PARAM1 OUT VARCHAR2 
) AS 
  CNT number(5);
BEGIN
  SELECT COUNT(*) INTO CNT FROM USERLIST WHERE UNAME=USERNAME AND PWD=PASSWORD;
  if(CNT<>0) then
  RESULT:="VALID-CREDETIALS";
  else
  RESULT:="INVALID-CREDENTIALS";
END IF;
  
END SPRING_AUTHENTICATION;

	"PWD" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	 CONSTRAINT "USERLIST_PK" PRIMARY KEY ("UNAME")
